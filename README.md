## 3.2 –ó–∞–ø—É—Å–∫ –¥–µ—Ç–µ–∫—Ü–∏–∏ –æ–¥–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è

# –ø—É—Ç—å –∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—é
image_path = ("data/Pictures/10.png") #—É–∫–∞–∑–∞—Ç—å –ø—É—Ç—å –≤–∞—à–µ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–ª–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–∑ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è

## 3.3 –ó–∞–ø—É—Å–∫ –¥–µ—Ç–µ–∫—Ü–∏–∏ –ø–∞–ø–∫–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π

```python
import cv2
from ultralytics import YOLO
import matplotlib.pyplot as plt
import os

model = YOLO("model/best.pt") 

# –ø–∞–ø–∫–∞ —Å –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏
folder_path = "data/Pictures" #—É–∫–∞–∑–∞—Ç—å –ø—É—Ç—å –≤–∞—à–µ–π –ø–∞–ø–∫–∏ —Å –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏ –∏–ª–∏ –ø—É—Ç—å –ø–∞–ø–∫–∏ –∏–∑ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è

# –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º—ã–µ —Ñ–æ—Ä–º–∞—Ç—ã
extensions = (".jpg", ".jpeg", ".png")

for filename in os.listdir(folder_path):
    if filename.lower().endswith(extensions):
        image_path = os.path.join(folder_path, filename)

        # —á–∏—Ç–∞–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
        img = cv2.imread(image_path)
        res = model.predict(img, conf=0.1, verbose=False)[0]

        print(f"\n==== {filename} ====")
        print("–ù–∞–π–¥–µ–Ω–Ω—ã–µ –æ–±—ä–µ–∫—Ç—ã:")

        for box in res.boxes:
            cls = int(box.cls)
            conf = float(box.conf)
            name = model.names[cls]
            print(f"- {name} ({conf:.2f})")

        # –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
        annotated = res.plot()

        # –ø–æ–∫–∞–∑–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç
        plt.imshow(cv2.cvtColor(annotated, cv2.COLOR_BGR2RGB))
        plt.axis('off')
        plt.show()
```
